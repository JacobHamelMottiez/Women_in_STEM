hughes.dion.data <- hughes.dion.data  |> mutate(lang = ifelse(univ %in% c('McGill', 'Concordia'), "Anglophone", "Francophone"))

hughes.dion.data <- hughes.dion.data  |> mutate(UQ = ifelse(univ %in% c('UQAM', 'UQO', 'UQAR', 'UQAT', 'UQTR', 'UQAC'), "UQ", "-UQ"))

univ.year <- hughes.dion.data |> select(univ, année, lang, UQ)
univ.year <- univ.year |> group_by(univ, année) |>  mutate(n = sum(année)) |> unique()

ggplot(univ.year, aes(x = année, y = n, group = univ, color = univ)) +
  geom_smooth() + 
  facet_wrap(facets = 'UQ')
